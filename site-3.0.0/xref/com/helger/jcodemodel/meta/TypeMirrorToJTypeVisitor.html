65  /**
66   * @author Victor Nazarov &lt;asviraspossible@gmail.com&gt;
67   */
68  class TypeMirrorToJTypeVisitor extends AbstractTypeVisitor6 <AbstractJType, Void>
69  {
70    private final ErrorTypePolicy m_aErrorTypePolicy;
71    private final TypeEnvironment m_aEnvironment;
72    private final JCodeModel m_aCodeModel;
73    private final DecidedErrorTypesModelsAdapter m_aModelsAdapter;
74  
75    public TypeMirrorToJTypeVisitor (final JCodeModel codeModel,
76                                     final DecidedErrorTypesModelsAdapter modelsAdapter,
77                                     final ErrorTypePolicy errorTypePolicy,
78                                     final TypeEnvironment environment)
79    {
80      this.m_aCodeModel = codeModel;
81      this.m_aModelsAdapter = modelsAdapter;
82      this.m_aErrorTypePolicy = errorTypePolicy;
83      this.m_aEnvironment = environment;
84    }
85  
86    @Override
87    public AbstractJType visitPrimitive (final PrimitiveType t, final Void p)
88    {
89      switch (t.getKind ())
90      {
91        case BOOLEAN:
92          return m_aCodeModel.BOOLEAN;
93        case BYTE:
94          return m_aCodeModel.BYTE;
95        case CHAR:
96          return m_aCodeModel.CHAR;
97        case INT:
98          return m_aCodeModel.INT;
99        case LONG:
100         return m_aCodeModel.LONG;
101       case FLOAT:
102         return m_aCodeModel.FLOAT;
103       case DOUBLE:
104         return m_aCodeModel.DOUBLE;
105       case SHORT:
106         return m_aCodeModel.SHORT;
107       default:
108         throw new IllegalArgumentException ("Unrecognized primitive " + t.getKind ());
109     }
110   }
111 
112   @Override
113   public AbstractJType visitNull (final NullType t, final Void p)
114   {
115     // To change body of generated methods, choose Tools | Templates.
116     throw new IllegalArgumentException ("null can't be JClass.");
117   }
118 
119   @Override
120   public AbstractJType visitArray (final ArrayType t, final Void p)
121   {
122     try
123     {
124       final AbstractJType componentType = m_aModelsAdapter.toJType (t.getComponentType (), m_aEnvironment);
125       return componentType.array ();
126     }
127     catch (final CodeModelBuildingException ex)
128     {
129       throw new RuntimeCodeModelBuildingException (ex);
130     }
131     catch (final ErrorTypeFound ex)
132     {
133       throw new RuntimeErrorTypeFound (ex);
134     }
135   }
136 
137   @Override
138   public AbstractJType visitDeclared (final DeclaredType t, final Void p)
139   {
140     try
141     {
142       final TypeElement element = (TypeElement) t.asElement ();
143       AbstractJClass _class = m_aModelsAdapter.ref (element);
144       for (final TypeMirror typeArgument : t.getTypeArguments ())
145       {
146         _class = _class.narrow (m_aModelsAdapter.toJType (typeArgument, m_aEnvironment));
147       }
148       return _class;
149     }
150     catch (final CodeModelBuildingException ex)
151     {
152       throw new RuntimeCodeModelBuildingException (ex);
153     }
154     catch (final ErrorTypeFound ex)
155     {
156       throw new RuntimeErrorTypeFound (ex);
157     }
158   }
159 
160   @Override
161   public AbstractJType visitError (final ErrorType t, final Void p)
162   {
163     final String typeName = t.asElement ().getSimpleName ().toString ();
164     final String fullTypeName = m_aEnvironment.packageName () + "." + typeName;
165     final JDefinedClass jCodeModelClass = m_aCodeModel._getClass (fullTypeName);
166     if (jCodeModelClass != null)
167     {
168       final List <? extends TypeMirror> typeArguments = t.getTypeArguments ();
169       if (typeArguments.isEmpty ())
170         return jCodeModelClass;
171       final List <AbstractJClass> jArguments = new ArrayList <> (typeArguments.size ());
172       for (final TypeMirror typeArgument : typeArguments)
173       {
174         try
175         {
176           jArguments.add ((AbstractJClass) m_aModelsAdapter.toJType (typeArgument, m_aEnvironment));
177         }
178         catch (final CodeModelBuildingException ex)
179         {
180           throw new RuntimeCodeModelBuildingException (ex);
181         }
182         catch (final ErrorTypeFound ex)
183         {
184           throw new RuntimeErrorTypeFound (ex);
185         }
186       }
187       return jCodeModelClass.narrow (jArguments);
188     }
189     if (m_aErrorTypePolicy.action () == ErrorTypePolicy.EAction.CREATE_ERROR_TYPE)
190       return m_aCodeModel.errorClass (typeName +
191                                     " in annotated source code",
192                                     typeName.equals ("<any>") ? null : typeName);
193     try
194     {
195       throw new ErrorTypeFound (typeName + " in annotated source code");
196     }
197     catch (final ErrorTypeFound ex)
198     {
199       throw new RuntimeErrorTypeFound (ex);
200     }
201   }
202 
203   @Override
204   public AbstractJType visitTypeVariable (final TypeVariable t, final Void p)
205   {
206     final String typeName = t.asElement ().getSimpleName ().toString ();
207     final AbstractJType result = m_aEnvironment.get (typeName);
208     if (result != null)
209       return result;
210     if (m_aErrorTypePolicy.action () == ErrorTypePolicy.EAction.CREATE_ERROR_TYPE)
211       return m_aCodeModel.errorClass ("Missing type-variable " + typeName + " in annotated source code");
212     try
213     {
214       throw new ErrorTypeFound ("Missing type-variable " + typeName + " in annotated source code");
215     }
216     catch (final ErrorTypeFound ex)
217     {
218       throw new RuntimeErrorTypeFound (ex);
219     }
220   }
221 
222   @Override
223   public AbstractJType visitWildcard (final WildcardType t, final Void p)
224   {
225     try
226     {
227       final TypeMirror extendsBoundMirror = t.getExtendsBound ();
228       if (extendsBoundMirror != null)
229       {
230         final AbstractJClass extendsBound = (AbstractJClass) m_aModelsAdapter.toJType (extendsBoundMirror, m_aEnvironment);
231         return extendsBound.wildcard (EWildcardBoundMode.EXTENDS);
232       }
233       final TypeMirror superBoundMirror = t.getSuperBound ();
234       if (superBoundMirror != null)
235       {
236         final AbstractJClass superBound = (AbstractJClass) m_aModelsAdapter.toJType (superBoundMirror, m_aEnvironment);
237         return superBound.wildcard (EWildcardBoundMode.SUPER);
238       }
239       return m_aCodeModel.wildcard ();
240     }
241     catch (final CodeModelBuildingException ex)
242     {
243       throw new RuntimeCodeModelBuildingException (ex);
244     }
245     catch (final ErrorTypeFound ex)
246     {
247       throw new RuntimeErrorTypeFound (ex);
248     }
249   }
250 
251   @Override
252   public AbstractJType visitExecutable (final ExecutableType t, final Void p)
253   {
254     throw new IllegalArgumentException ("executable can't be JClass.");
255   }
256 
257   @Override
258   public AbstractJType visitNoType (final NoType t, final Void p)
259   {
260     if (m_aErrorTypePolicy.action () == ErrorTypePolicy.EAction.CREATE_ERROR_TYPE)
261       return m_aCodeModel.errorClass ("'no type' in annotated source code");
262 
263     try
264     {
265       throw new ErrorTypeFound ("'no type' in annotated source code");
266     }
267     catch (final ErrorTypeFound ex)
268     {
269       throw new RuntimeErrorTypeFound (ex);
270     }
271   }
272 
273   @Override
274   public AbstractJType visitUnknown (final TypeMirror t, final Void p)
275   {
276     if (m_aErrorTypePolicy.action () == ErrorTypePolicy.EAction.CREATE_ERROR_TYPE)
277       return m_aCodeModel.errorClass ("'unknown type' in annotated source code");
278 
279     try
280     {
281       throw new ErrorTypeFound ("'unknown type' in annotated source code");
282     }
283     catch (final ErrorTypeFound ex)
284     {
285       throw new RuntimeErrorTypeFound (ex);
286     }
287   }
