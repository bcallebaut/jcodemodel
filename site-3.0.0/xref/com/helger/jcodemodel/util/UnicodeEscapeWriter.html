41  package com.helger.jcodemodel.util;
42  
43  import java.io.FilterWriter;
44  import java.io.IOException;
45  import java.io.Writer;
46  import java.util.BitSet;
47  
48  import javax.annotation.Nonnegative;
49  import javax.annotation.Nonnull;
50  
51  /**
52   * {@link Writer} that escapes non US-ASCII characters into Java Unicode escape
53   * \\uXXXX. This process is necessary if the method names or field names contain
54   * non US-ASCII characters.
55   *
56   * @author Kohsuke Kawaguchi (kohsuke.kawaguchi@sun.com)
57   */
58  public class UnicodeEscapeWriter extends FilterWriter
59  {
60    private static final BitSet ESCAPE = new BitSet (128);
61  
62    static
63    {
64      for (int i = 0; i < 0x20; i++)
65        if (i != '\t' && i != '\r' && i != '\n')
66          ESCAPE.set (i, true);
67    }
68  
69    public UnicodeEscapeWriter (@Nonnull final Writer aNext)
70    {
71      super (aNext);
72    }
73  
74    @Override
75    public final void write (final int ch) throws IOException
76    {
77      if (requireEscaping (ch))
78      {
79        // need to escape
80        out.write ("\\u");
81        final String s = Integer.toHexString (ch);
82        for (int i = s.length (); i < 4; i++)
83          out.write ('0');
84        out.write (s);
85      }
86      else
87        out.write (ch);
88    }
89  
90    /**
91     * Can be overridden.
92     *
93     * @param ch
94     *        Character to check
95     * @return true if the character needs to be escaped.
96     */
97    protected boolean requireEscaping (final int ch)
98    {
99      // control characters
100     return ch >= 128 || ESCAPE.get (ch);
101   }
102 
103   @Override
104   public final void write (@Nonnull final char [] aBuf,
105                            @Nonnegative final int nOfs,
106                            @Nonnegative final int nLen) throws IOException
107   {
108     for (int i = 0; i < nLen; i++)
109       write (aBuf[nOfs + i]);
110   }
111 
112   @Override
113   public final void write (@Nonnull final char [] aBuf) throws IOException
114   {
115     write (aBuf, 0, aBuf.length);
116   }
117 
118   @Override
119   public final void write (@Nonnull final String sStr,
120                            @Nonnegative final int nOfs,
121                            @Nonnegative final int nLen) throws IOException
122   {
123     write (sStr.toCharArray (), nOfs, nLen);
124   }
125 
126   @Override
127   public final void write (@Nonnull final String sStr) throws IOException
128   {
129     write (sStr.toCharArray (), 0, sStr.length ());
130   }
131 }
