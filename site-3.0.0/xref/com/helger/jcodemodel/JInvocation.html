141   private JInvocation (@Nullable final JCodeModel aOwner,
142                        @Nullable final IJGenerable aObject,
143                        @Nonnull final String sName)
144   {
145     JCValueEnforcer.notNull (sName, "Name");
146     JCValueEnforcer.isFalse (sName.indexOf ('.') >= 0, () -> "method name contains '.': " + sName);
147     m_aOwner = aOwner;
148     m_aObject = aObject;
149     m_sMethodName = sName;
150     m_aMethod = null;
151     m_bIsConstructor = false;
152     m_aConstructorType = null;
153   }

155   private JInvocation (@Nonnull final JCodeModel aOwner,
156                        @Nullable final IJGenerable aObject,
157                        @Nonnull final JMethod aMethod)
158   {
159     JCValueEnforcer.notNull (aOwner, "Owner");
160     JCValueEnforcer.notNull (aMethod, "Method");
161     m_aOwner = aOwner;
162     m_aObject = aObject;
163     m_sMethodName = null;
164     m_aMethod = aMethod;
165     m_bIsConstructor = false;
166     m_aConstructorType = null;
167   }

173    *        Type of the object to be created. If this type is an array type,
174    *        added arguments are treated as array initializer. Thus you can
175    *        create an expression like <code>new int[]{1,2,3,4,5}</code>.
176    */
177   protected JInvocation (@Nonnull final AbstractJType aConstructorType)
178   {
179     JCValueEnforcer.notNull (aConstructorType, "ConstructorType");
180     m_aOwner = aConstructorType.owner ();
181     m_aObject = null;
182     m_sMethodName = null;
183     m_aMethod = null;
184     m_bIsConstructor = true;
185     m_aConstructorType = aConstructorType;
186   }

329   @Nonnull
330   private JCodeModel _narrowOwner ()
331   {
332     final JCodeModel owner = owner ();
333     if (owner == null)
334       throw new IllegalStateException ("No owner is present, so this invocation cannot be generified!");
335     return owner;
336   }

339   public JInvocation narrow (@Nonnull final String sName)
340   {
341     final JTypeVar v = new JTypeVar (_narrowOwner (), sName);
342     if (m_aTypeVariables == null)
343       m_aTypeVariables = new ArrayList <> (3);
344     m_aTypeVariables.add (v);

372   private void _addTypeVars (@Nonnull final JFormatter f)
373   {
374     if (m_aTypeVariables != null && !m_aTypeVariables.isEmpty ())
375     {
376       f.print ('<');
377       int nIndex = 0;
378       for (final JTypeVar aTypeVar : m_aTypeVariables)
379       {
380         if (nIndex++ > 0)
381           f.print (',');
382         // Use type here to get the import (if needed)
383         f.type (aTypeVar);
384       }
385       f.print (JFormatter.CLOSE_TYPE_ARGS);
386     }
387   }

392     return m_aMethod != null ? m_aMethod.name () : m_sMethodName;

395   public void generate (@Nonnull final JFormatter f)
396   {
397     if (m_bIsConstructor)
398     {
399       if (m_aConstructorType.isArray ())
400       {
401         // [RESULT] new T[]{arg1,arg2,arg3,...};
402         f.print ("new").generable (m_aConstructorType);
403         _addTypeVars (f);
404         f.print ('{');
405       }
406       else
407       {
408         // [RESULT] new T(
409         f.print ("new").generable (m_aConstructorType);
410         _addTypeVars (f);
411         f.print ('(');
412       }
413     }
414     else
415     {
416       // Not a constructor
417       final String name = _methodName ();
418 
419       if (m_aObject != null)
420       {
421         // object.<generics> name (
422         f.generable (m_aObject).print ('.');
423         _addTypeVars (f);
424         f.print (name);
425         f.print ('(');
426       }
427       else
428       {
429         // name (
430         f.id (name).print ('(');
431       }
432     }
433 
434     // Method arguments
435     f.generable (m_aArgs);
436 
437     // Close arg list
438     if (m_bIsConstructor && m_aConstructorType.isArray ())
439       f.print ('}');
440     else
441       f.print (')');
442 
443     if (m_aConstructorType instanceof JDefinedClass && ((JDefinedClass) m_aConstructorType).isAnonymous ())
444     {
445       ((JAnonymousClass) m_aConstructorType).declareBody (f);
446     }
447   }

460   @Override
461   public boolean equals (final Object o)
462   {
463     if (o == this)
464       return true;
465     if (o == null || getClass () != o.getClass ())
466       return false;
467     final JInvocation rhs = (JInvocation) o;
468     if (!(isEqual (m_aObject, rhs.m_aObject) &&
469           isEqual (m_bIsConstructor, rhs.m_bIsConstructor) &&
470           (m_bIsConstructor || isEqual (_methodName (), rhs._methodName ())) &&
471           isEqual (m_aArgs, rhs.m_aArgs) &&
472           isEqual (_typeFullName (), rhs._typeFullName ())))
473     {
474       return false;
475     }
476     if (m_aTypeVariables == null)
477       return rhs.m_aTypeVariables == null;
478     if (rhs.m_aTypeVariables == null)
479       return false;
480     if (m_aTypeVariables.size () != rhs.m_aTypeVariables.size ())
481       return false;
482     for (int i = 0; i < m_aTypeVariables.size (); i++)
483     {
484       if (!isEqual (m_aTypeVariables.get (i).fullName (), rhs.m_aTypeVariables.get (i).fullName ()))
485         return false;
486     }
487     return true;
488   }

490   @Override
491   public int hashCode ()
492   {
493     JCHashCodeGenerator hashCodeGenerator = new JCHashCodeGenerator (this).append (m_aObject).append (m_bIsConstructor);
494     if (!m_bIsConstructor)
495       hashCodeGenerator = hashCodeGenerator.append (_methodName ());
496     hashCodeGenerator = hashCodeGenerator.append (m_aArgs).append (_typeFullName ());
497     if (m_aTypeVariables != null)
498     {
499       hashCodeGenerator = hashCodeGenerator.append (m_aTypeVariables.size ());
500       for (final JTypeVar typeVariable : m_aTypeVariables)
501       {
502         hashCodeGenerator = hashCodeGenerator.append (typeVariable.fullName ());
503       }
504     }
505     return hashCodeGenerator.getHashCode ();
506   }
