3   import javax.annotation.CheckForSigned;
4   import javax.annotation.Nonnegative;
5   import javax.annotation.Nonnull;
6   import javax.annotation.Nullable;

8   public final class JCStringHelper
9   {
10    /** Constant empty String array */
11    public static final String [] EMPTY_STRING_ARRAY = new String [0];
12  
13    private JCStringHelper ()
14    {}
15  
16    /**
17     * Check if the string is <code>null</code> or empty.
18     *
19     * @param sStr
20     *        The string to check. May be <code>null</code>.
21     * @return <code>true</code> if the string is <code>null</code> or empty,
22     *         <code>false</code> otherwise
23     */
24    public static boolean hasNoText (@Nullable final String sStr)
25    {
26      return sStr == null || sStr.isEmpty ();
27    }
28  
29    @Nonnegative
30    public static int getCharCount (@Nullable final String s, final char cSearch)
31    {
32      return s == null ? 0 : getCharCount (s.toCharArray (), cSearch);
33    }
34  
35    @Nonnegative
36    public static int getCharCount (@Nullable final char [] aChars, final char cSearch)
37    {
38      int ret = 0;
39      if (aChars != null)
40        for (final char c : aChars)
41          if (c == cSearch)
42            ++ret;
43      return ret;
44    }
45  
46    /**
47     * Take a concatenated String and return the passed String array of all
48     * elements in the passed string, using specified separator char.
49     *
50     * @param cSep
51     *        The separator to use.
52     * @param sElements
53     *        The concatenated String to convert. May be <code>null</code> or
54     *        empty.
55     * @return The passed collection and never <code>null</code>.
56     */
57    @Nonnull
58    public static String [] getExplodedArray (final char cSep, @Nullable final String sElements)
59    {
60      return getExplodedArray (cSep, sElements, -1);
61    }
62  
63    /**
64     * Take a concatenated String and return the passed String array of all
65     * elements in the passed string, using specified separator char.
66     *
67     * @param cSep
68     *        The separator to use.
69     * @param sElements
70     *        The concatenated String to convert. May be <code>null</code> or
71     *        empty.
72     * @param nMaxItems
73     *        The maximum number of items to explode. If the passed value is &le;
74     *        0 all items are used. If max items is 1, than the result string is
75     *        returned as is. If max items is larger than the number of elements
76     *        found, it has no effect.
77     * @return The passed collection and never <code>null</code>.
78     */
79    @Nonnull
80    public static String [] getExplodedArray (final char cSep,
81                                              @Nullable final String sElements,
82                                              @CheckForSigned final int nMaxItems)
83    {
84      if (nMaxItems == 1)
85        return new String [] { sElements };
86      if (hasNoText (sElements))
87        return EMPTY_STRING_ARRAY;
88  
89      final int nMaxResultElements = 1 + getCharCount (sElements, cSep);
90      if (nMaxResultElements == 1)
91      {
92        // Separator not found
93        return new String [] { sElements };
94      }
95      final String [] ret = new String [nMaxItems < 1 ? nMaxResultElements : Math.min (nMaxResultElements, nMaxItems)];
96  
97      // Do not use RegExCache.stringReplacePattern because of package
98      // dependencies
99      // Do not use String.split because it trims empty tokens from the end
100     int nStartIndex = 0;
101     int nItemsAdded = 0;
102     while (true)
103     {
104       final int nMatchIndex = sElements.indexOf (cSep, nStartIndex);
105       if (nMatchIndex < 0)
106         break;
107 
108       ret[nItemsAdded++] = sElements.substring (nStartIndex, nMatchIndex);
109       // 1 == length of separator char
110       nStartIndex = nMatchIndex + 1;
111       if (nMaxItems > 0 && nItemsAdded == nMaxItems - 1)
112       {
113         // We have exactly one item the left: the rest of the string
114         break;
115       }
116     }
117     ret[nItemsAdded++] = sElements.substring (nStartIndex);
118     if (nItemsAdded != ret.length)
119       throw new IllegalStateException ("Added " + nItemsAdded + " but expected " + ret.length);
120     return ret;
121   }
122 }
