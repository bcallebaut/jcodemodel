428   /**
429    * Escapes the given string, then surrounds it by the specified quotation
430    * mark.
431    *
432    * @param cQuote
433    *        Quote char. Either single quote (') or double quote (")
434    * @param sStr
435    *        Source string to quote
436    * @return Qutoed string
437    */
438   @Nonnull
439   public static String quotify (final char cQuote, @Nonnull final String sStr)
440   {
441     final int n = sStr.length ();
442     final StringBuilder sb = new StringBuilder (n + 2);
443     sb.append (cQuote);
444     for (final char c : sStr.toCharArray ())
445     {
446       final int j = CHAR_ESCAPE.indexOf (c);
447       if (j >= 0)
448       {
449         if ((cQuote == '"' && c == '\'') || (cQuote == '\'' && c == '"'))
450         {
451           sb.append (c);
452         }
453         else
454         {
455           sb.append ('\\');
456           sb.append (CHAR_MACRO.charAt (j));
457         }
458       }
459       else
460       {
461         // technically Unicode escape shouldn't be done here,
462         // for it's a lexical level handling.
463         //
464         // However, various tools are so broken around this area,
465         // so just to be on the safe side, it's better to do
466         // the escaping here (regardless of the actual file encoding)
467         //
468         // see bug
469         if (c < 0x20 || c > 0x7E)
470         {
471           // not printable. use Unicode escape
472           sb.append ("\\u");
473           final String hex = Integer.toHexString (c & 0xFFFF);
474           for (int k = hex.length (); k < 4; k++)
475             sb.append ('0');
476           sb.append (hex);
477         }
478         else
479         {
480           sb.append (c);
481         }
482       }
483     }
484     sb.append (cQuote);
485     return sb.toString ();
486   }

500   /**
501    * Creates an aExpr directly from a source code fragment.
502    * <p>
503    * This method can be used as a short-cut to create a JExpression. For
504    * example, instead of <code>_a.gt(_b)</code>, you can write it as:
505    * <code>JExpr.direct("a&gt;b")</code>.
506    * <p>
507    * Be warned that there is a danger in using this method, as it obfuscates the
508    * object model.
509    *
510    * @param sSourceCode
511    *        Java source code
512    * @return Direct aExpr
513    */
514   @Nonnull
515   public static IJExpression direct (@Nonnull final String sSourceCode)
516   {
517     return f -> f.print ('(').print (sSourceCode).print (')');
518   }
