41  package com.helger.jcodemodel;
42  
43  import java.util.ArrayList;
44  import java.util.List;
45  
46  import javax.annotation.Nonnull;
47  
48  /**
49   * Try statement with Catch and/or Finally clause.
50   */
51  public class JTryBlock implements IJStatement
52  {
53    private final List <JTryResource> m_aResources = new ArrayList <> ();
54    private final JBlockl#JBlock">JBlock m_aBody = new JBlock ();
55    private final List <JCatchBlock> m_aCatches = new ArrayList <> ();
56    private JBlock m_aFinally;
57  
58    public JTryBlock ()
59    {}
60  
61    /**
62     * @return A mutable list of all try-resources for "try-with-resources".
63     * @since 3.2.3
64     */
65    @Nonnull
66    public List <JTryResource> tryResources ()
67    {
68      return m_aResources;
69    }
70  
71    /**
72     * @return The non-<code>null</code> try-body.
73     */
74    @Nonnull
75    public JBlock body ()
76    {
77      return m_aBody;
78    }
79  
80    /**
81     * Add a new catch block for the specification exception class. The variable
82     * name is automatically created.
83     *
84     * @param aException
85     *        The exception class to catch. May not be <code>null</code>.
86     * @return the created catch block for further customization and never
87     *         <code>null</code>.
88     * @see #catches()
89     */
90    @Nonnull
91    public JCatchBlock _catch (@Nonnull final AbstractJClass aException)
92    {
93      final JCatchBlockk.html#JCatchBlock">JCatchBlock cb = new JCatchBlock (aException);
94      m_aCatches.add (cb);
95      return cb;
96    }
97  
98    /**
99     * Get a list of all catch blocks. Since v3.2.3 the returned list is mutable.
100    * Previously it was immutable.
101    *
102    * @return A mutable list of all contained catch blocks.
103    */
104   @Nonnull
105   public List <JCatchBlock> catches ()
106   {
107     return m_aCatches;
108   }
109 
110   /**
111    * @return <code>true</code> if at least one catch block is present,
112    *         <code>false</code> if not.
113    * @since 3.2.3
114    */
115   public boolean hasCatchBlock ()
116   {
117     return !m_aCatches.isEmpty ();
118   }
119 
120   /**
121    * @return A non-<code>null</code> finally block. The block is automatically
122    *         created the first time you call this method.
123    */
124   @Nonnull
125   public JBlock _finally ()
126   {
127     if (m_aFinally == null)
128       m_aFinally = new JBlock ();
129     return m_aFinally;
130   }
131 
132   /**
133    * Remove the finally block - this allows to reset an eventually accidentally
134    * created finally block.
135    *
136    * @since 3.2.3
137    */
138   public void resetFinally ()
139   {
140     m_aFinally = null;
141   }
142 
143   /**
144    * @return <code>true</code> if a finally block is present, <code>false</code>
145    *         if not.
146    * @since 3.2.3
147    */
148   public boolean hasFinally ()
149   {
150     return m_aFinally != null;
151   }
152 
153   public void state (@Nonnull final IJFormatter f)
154   {
155     f.print ("try");
156     if (!m_aResources.isEmpty ())
157     {
158       f.print ('(');
159       boolean bFirst = true;
160       for (final JTryResource aResource : m_aResources)
161       {
162         if (bFirst)
163           bFirst = false;
164         else
165           f.print (';').newline ();
166         f.generable (aResource);
167       }
168       f.print (')');
169     }
170     f.generable (m_aBody);
171     for (final JCatchBlock cb : m_aCatches)
172       f.generable (cb);
173     if (m_aFinally != null)
174       f.print ("finally").generable (m_aFinally);
175     f.newline ();
176   }
177 }
