74  /**
75   * Root of the code DOM.
76   * <p>
77   * Here's your typical CodeModel application.
78   *
79   * <pre>
80   * JCodeModel cm = new JCodeModel();
81   *
82   * // generate source code by populating the 'cm' tree.
83   * cm._class(...);
84   * ...
85   *
86   * // write them out
87   * cm.build(new File("."));
88   * </pre>
89   * <p>
90   * Every CodeModel node is always owned by one {@link JCodeModel} object at any
91   * given time (which can be often accessed by the <tt>owner()</tt> method.) As
92   * such, when you generate Java code, most of the operation works in a top-down
93   * fashion. For example, you create a class from {@link JCodeModel}, which gives
94   * you a {@link JDefinedClass}. Then you invoke a method on it to generate a new
95   * method, which gives you {@link JMethod}, and so on. There are a few
96   * exceptions to this, most notably building {@link IJExpression}s, but
97   * generally you work with CodeModel in a top-down fashion. Because of this
98   * design, most of the CodeModel classes aren't directly instanciable.
99   * <h2>Where to go from here?</h2>
100  * <p>
101  * Most of the time you'd want to populate new type definitions in a
102  * {@link JCodeModel}. See {@link #_class(String, EClassType)}.
103  */
104 public class JCodeModel implements Serializable

106   protected static boolean checkIfFileSystemIsCaseSensitive ()
107   {
108     try
109     {
110       // let the system property override, in case the user really
111       // wants to override.
112       if (System.getProperty ("com.sun.codemodel.FileSystemCaseSensitive") != null)
113         return true;
114 
115       // Add special override to differentiate if Sun implementation is also in
116       // scope
117       if (System.getProperty ("com.helger.jcodemodel.FileSystemCaseSensitive") != null)
118         return true;
119     }
120     catch (final Exception e)
121     {
122       // Fall through
123     }
124 
125     // on Unix, it's case sensitive.
126     return File.separatorChar == '/';
127   }

176     primitiveToBox = Collections.unmodifiableMap (m2);

371    * Creates a dummy, error {@link AbstractJClass} that can only be referenced
372    * from hidden classes.
373    * <p>
374    * This method is useful when the code generation needs to include some error
375    * class that should never leak into actually written code.
376    * <p>
377    * Error-types represents holes or place-holders that can't be filled.
378    * References to error-classes can be used in hidden class-models. Such classes
379    * should never be actually written but can be somehow used during code
380    * generation. Use {@code JCodeModel#buildsErrorTypeRefs} method to test if your
381    * generated Java-sources contains references to error-types.
382    * <p>
383    * You should probably always check generated code with
384    * {@code JCodeModel#buildsErrorTypeRefs} method if you use any error-types.
385    * <p>
386    * Most of error-types methods throws {@code JErrorClassUsedException} unchecked
387    * exceptions. Be careful and use {@link AbstractJType#isError()
388    * AbstractJType#isError} method to check for error-types before actually using
389    * it's methods.
390    *
391    * @param sMessage
392    *        some free form text message to identify source of error
393    * @return New {@link JErrorClass}
394    * @see JCodeModel#buildsErrorTypeRefs()
395    * @see JErrorClass
396    */

907   @NotThreadSafe
908   private final class TypeNameParser
909   {
910     private final String m_sTypeName;
911     private int m_nIdx;
912 
913     public TypeNameParser (@Nonnull final String sTypeName)
914     {
915       m_sTypeName = sTypeName;
916     }
917 
918     /**
919      * Parses a type name token T (which can be potentially of the form
920      * T&lt;T1,T2,...&gt;, or "? extends/super T".)
921      *
922      * @return The parsed type name
923      */
924     @Nonnull
925     AbstractJClass parseTypeName ()
926     {
927       final int nStart = m_nIdx;
928 
929       if (m_sTypeName.charAt (m_nIdx) == '?')
930       {
931         // wildcard
932         m_nIdx++;
933         _skipWs ();
934 
935         final String head = m_sTypeName.substring (m_nIdx);
936         if (head.startsWith ("extends"))
937         {
938           // 7 == "extends".length
939           m_nIdx += 7;
940           _skipWs ();
941           return parseTypeName ().wildcardExtends ();
942         }
943 
944         if (head.startsWith ("super"))
945         {
946           // 5 == "super".length
947           m_nIdx += 5;
948           _skipWs ();
949           return parseTypeName ().wildcardSuper ();
950         }
951 
952         // not supported
953         throw new IllegalArgumentException ("only extends/super can follow ?, but found " +
954                                             m_sTypeName.substring (m_nIdx));
955       }
956 
957       while (m_nIdx < m_sTypeName.length ())
958       {
959         final char ch = m_sTypeName.charAt (m_nIdx);
960         if (Character.isJavaIdentifierStart (ch) || Character.isJavaIdentifierPart (ch) || ch == '.')
961           m_nIdx++;
962         else
963           break;
964       }
965 
966       final AbstractJClass aClazz = ref (m_sTypeName.substring (nStart, m_nIdx));
967       return _parseSuffix (aClazz);
968     }
969 
970     /**
971      * Parses additional left-associative suffixes, like type arguments and array
972      * specifiers.
973      */
974     @Nonnull
975     private AbstractJClasss">AbstractJClass _parseSuffix (@Nonnull final AbstractJClass aClazz)
976     {
977       if (m_nIdx == m_sTypeName.length ())
978       {
979         // hit EOL
980         return aClazz;
981       }
982 
983       final char ch = m_sTypeName.charAt (m_nIdx);
984 
985       if (ch == '<')
986         return _parseSuffix (_parseArguments (aClazz));
987 
988       if (ch == '[')
989       {
990         if (m_sTypeName.charAt (m_nIdx + 1) == ']')
991         {
992           m_nIdx += 2;
993           return _parseSuffix (aClazz.array ());
994         }
995         throw new IllegalArgumentException ("Expected ']' but found " + m_sTypeName.substring (m_nIdx + 1));
996       }
997 
998       return aClazz;
999     }
1000 
1001     /**
1002      * Skips whitespaces
1003      */
1004     private void _skipWs ()
1005     {
1006       while (Character.isWhitespace (m_sTypeName.charAt (m_nIdx)) && m_nIdx < m_sTypeName.length ())
1007         m_nIdx++;
1008     }
1009 
1010     /**
1011      * Parses '&lt;T1,T2,...,Tn&gt;'
1012      *
1013      * @return the index of the character next to '>'
1014      */
1015     @Nonnull
1016     private AbstractJClassAbstractJClass _parseArguments (@Nonnull final AbstractJClass aRawType)
1017     {
1018       JCValueEnforcer.isTrue (m_sTypeName.charAt (m_nIdx) == '<', "Expected '<' at current index");
1019       m_nIdx++;
1020 
1021       final List <AbstractJClass> args = new ArrayList <> ();
1022 
1023       while (true)
1024       {
1025         args.add (parseTypeName ());
1026         if (m_nIdx == m_sTypeName.length ())
1027           throw new IllegalArgumentException ("Missing '>' in " + m_sTypeName);
1028         final char ch = m_sTypeName.charAt (m_nIdx);
1029         if (ch == '>')
1030           return aRawType.narrow (args);
1031 
1032         if (ch != ',')
1033           throw new IllegalArgumentException (m_sTypeName);
1034         m_nIdx++;
1035       }
1036     }
