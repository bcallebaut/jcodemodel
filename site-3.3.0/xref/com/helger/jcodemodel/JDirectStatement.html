41  package com.helger.jcodemodel;
42  
43  import javax.annotation.Nonnull;
44  
45  import com.helger.jcodemodel.util.JCValueEnforcer;
46  
47  /**
48   * This class represents a single direct statement. A direct statement is a
49   * statement that is neither parsed not evaluated! Handle with care!
50   *
51   * @author Philip Helger
52   * @since 2.7.10
53   */
54  public final class JDirectStatement implements IJStatement
55  {
56    private String m_sSource;
57  
58    public JDirectStatement (@Nonnull final String sSource)
59    {
60      source (sSource);
61    }
62  
63    @Nonnull
64    public String source ()
65    {
66      return m_sSource;
67    }
68  
69    public void source (@Nonnull final String sSource)
70    {
71      JCValueEnforcer.notNull (sSource, "Source");
72      m_sSource = sSource;
73    }
74  
75    public void state (@Nonnull final IJFormatter f)
76    {
77      f.print (m_sSource).newline ();
78    }
79  }
