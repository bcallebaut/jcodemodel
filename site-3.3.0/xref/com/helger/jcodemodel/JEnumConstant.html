56  /**
57   * Enum Constant. When used as an {@link IJExpression}, this object represents a
58   * reference to the enum constant.

60   * @author Bhakti Mehta (Bhakti.Mehta@sun.com)

62  public class JEnumConstant implements IJExpression, IJDeclaration, IJAnnotatable, IJDocCommentable
63  {
64    /**
65     * The enum class.
66     */
67    private final AbstractJClass m_aType;
68  
69    /**
70     * The constant.

72    private final String m_sName;
73  
74    /**
75     * javadoc comments, if any.

77    private JDocComment m_aJavaDoc;

79    /**

117   @Nonnull
118   public JEnumConstant arg (@Nonnull final IJExpression aArg)
119   {
120     JCValueEnforcer.notNull (aArg, "Arg");
121     if (m_aArgs == null)

140     if (m_aArgs == null)

145   public boolean hasArgs ()

148   }

150   /**
151    * Returns the name of this constant including the type name
152    *
153    * @return never null.
154    */

156   public String getName ()
157   {
158     return m_aType.fullName () + '.' + m_sName;
159   }

163   {

165       m_aJavaDoc = new JDocComment (m_aType.owner ());
166     return m_aJavaDoc;
167   }

202   {
203     if (m_aAnnotations == null)
204       m_aAnnotations = new ArrayList <> ();
205     return Collections.unmodifiableList (m_aAnnotations);
206   }

208   public void declare (@Nonnull final IJFormatter f)
209   {
210     if (m_aJavaDoc != null)
211       f.newline ().generable (m_aJavaDoc);
212     if (m_aAnnotations != null)
213       for (final JAnnotationUse annotation : m_aAnnotations)
214         f.generable (annotation).newline ();
215     f.id (m_sName);
216     if (m_aArgs != null)
217       f.print ('(').generable (m_aArgs).print (')');
218   }
219 
220   public void generate (@Nonnull final IJFormatter f)
221   {
222     f.type (m_aType).print ('.').print (m_sName);
223   }
