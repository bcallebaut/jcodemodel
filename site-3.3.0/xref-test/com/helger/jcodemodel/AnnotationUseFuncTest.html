41  package com.helger.jcodemodel;
42  
43  import static org.junit.Assert.assertEquals;
44  
45  import java.lang.annotation.Retention;
46  import java.lang.annotation.RetentionPolicy;
47  import java.lang.annotation.Target;
48  
49  import org.junit.Test;
50  
51  import com.helger.jcodemodel.util.CodeModelTestsHelper;
52  
53  /**
54   * A test program for the annotation use features Note: Not all the generated
55   * code would make sense but just checking in all the different ways you can use
56   * an annotation
57   *
58   * @author Bhakti Mehta
59   */
60  public final class AnnotationUseFuncTest
61  {
62    @interface XmlElement
63    {
64      String value();
65  
66      String ns();
67    }
68  
69    interface XmlElementW extends IJAnnotationWriter <XmlElement>
70    {
71      XmlElementW value (String s);
72  
73      XmlElementW ns (String s);
74    }
75  
76    /**
77     * *********************************************************************
78     * Generates this
79     * **********************************************************************
80     *
81     * <pre>
82     * import java.lang.annotation.Retention;
83     * import java.lang.annotation.RetentionPolicy;
84     * import java.lang.annotation.Target;
85     * import com.helger.jcodemodel.tests.AnnotationUseTest;
86     *
87     * &#064;Retention (value = Test.Iamenum.GOOD, value1 = RetentionPolicy.RUNTIME)
88     * &#064;AnnotationUseTest.XmlElement (ns = &quot;##default&quot;, value = &quot;foobar&quot;)
89     * public class Test
90     * {
91     *   &#064;Retention (name = &quot;book&quot;,
92     *               targetNamespace = 5,
93     *               names = { &quot;Bob&quot;, &quot;Rob&quot;, &quot;Ted&quot; },
94     *               namesno = { 4, 5, 6 },
95     *               values = { @Target (type = java.lang.Integer.class), &#64;Target (type = java.lang.Float.class) },
96     *               foo = @Target (junk = 7))
97     *   private double y;
98     *
99     *   public void foo ()
100    *   {}
101    *
102    *   public enum Iamenum
103    *   {
104    *     GOOD,
105    *     BAD;
106    *   }
107    * }
108    * </pre>
109    *
110    * @throws JClassAlreadyExistsException
111    *         internally only
112    */
113   @Test
114   public void testMain () throws JClassAlreadyExistsException
115   {
116     final JCodeModel cm = new JCodeModel ();
117     final JDefinedClass cls = cm._class ("Test");
118     // JMethod m =
119     cls.method (JMod.PUBLIC, cm.VOID, "foo");
120 
121     // Annotating a class
122     // Using the existing Annotations from java.lang.annotation package
123     final JAnnotationUse use = cls.annotate (cm.ref (Retention.class));
124 
125     // declaring an enum class and an enumconstant as a membervaluepair
126     final JDefinedClass enumcls = cls._enum ("Iamenum");
127     final JEnumConstant ec = enumcls.enumConstant ("GOOD");
128     final JEnumConstant ec1 = enumcls.enumConstant ("BAD");
129     final JEnumConstant ec2 = enumcls.enumConstant ("BAD");
130     assertEquals (ec1, ec2);
131 
132     use.param ("value", ec);
133     // adding another param as an enum
134     use.param ("value1", RetentionPolicy.RUNTIME);
135 
136     // Adding annotation for fields
137     // will generate like
138     // @String(name = "book") private double y;
139     //
140     final JFieldVar field = cls.field (JMod.PRIVATE, cm.DOUBLE, "y");
141 
142     // Adding more annotations which are member value pairs
143     final JAnnotationUse aUse = field.annotate (Retention.class);
144     aUse.param ("name", "book");
145     aUse.param ("targetNamespace", 5);
146 
147     // Adding arrays as member value pairs
148     final JAnnotationArrayMember arrayMember = aUse.paramArray ("names");
149     arrayMember.param ("Bob");
150     arrayMember.param ("Rob");
151     arrayMember.param ("Ted");
152 
153     // Shortcut
154     // Ambiguous on OpenJDK6 - explicit array needed
155     aUse.paramArray ("namesno", new int [] { 4, 5, 6 });
156 
157     final JAnnotationArrayMember arrayMember2 = aUse.paramArray ("values");
158     // adding an annotation as a member value pair
159     arrayMember2.annotate (Target.class).param ("type", Integer.class);
160     arrayMember2.annotate (Target.class).param ("type", Float.class);
161 
162     // test typed annotation writer
163     final XmlElementW w = cls.annotate2 (XmlElementW.class);
164     w.ns ("##default").value ("foobar");
165 
166     // adding an annotation as a member value pair
167     final JAnnotationUse myuse = aUse.annotationParam ("foo", Target.class);
168     myuse.param ("junk", 7);
169     myuse.param ("value-value");
170 
171     CodeModelTestsHelper.parseCodeModel (cm);
172   }
