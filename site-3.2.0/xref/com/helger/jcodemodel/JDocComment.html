51  /**
52   * JavaDoc comment.
53   * <p>
54   * A javadoc comment consists of multiple parts. There's the main part (that
55   * comes the first in in the comment section), then the parameter parts
56   * (@param), the return part (@return), and the throws parts (@throws).<br>
57   * TODO: it would be nice if we have JComment class and we can derive this class
58   * from there.
59   */
60  public class JDocComment extends JCommentPart implements IJGenerable, IJOwned
61  {
62    public static final String TAG_AUTHOR = "author";
63    public static final String TAG_DEPRECATED = "deprecated";
64    public static final String TAG_SEE = "see";
65    public static final String TAG_SINCE = "since";
66    public static final String TAG_VERSION = "version";
67  
68    private static final long serialVersionUID = 1L;
69  
70    private final JCodeModel m_aOwner;
71  
72    private boolean m_bIsSingleLineMode = false;
73  
74    /**
75     * list of @param tags
76     */
77    private final Map <String, JCommentPart> m_aAtParams = new LinkedHashMap <> ();
78  
79    /**
80     * The @return tag part.
81     */
82    private JCommentPart m_aAtReturn;
83  
84    /**
85     * list of @throws tags
86     */
87    private final Map <AbstractJClass, JCommentPart> m_aAtThrows = new LinkedHashMap <> ();
88  
89    /**
90     * Other comment tags (like @author, @deprecated, @since, @version etc.)
91     */
92    private final Map <String, JCommentPart> m_aAtTags = new LinkedHashMap <> ();
93  
94    /** list of generic xdoclets */
95    private final Map <String, Map <String, String>> m_aAtXdoclets = new LinkedHashMap <> ();
96  
97    protected JDocComment (@Nonnull final JCodeModel owner)

333   @Nonnull
334   public Map <String, String> addXdoclet (@Nonnull final String sName)

336     JCValueEnforcer.notNull (sName, "Name");
337     return m_aAtXdoclets.computeIfAbsent (sName, k -> new LinkedHashMap <> ());
338   }

340   /**
341    * add an xdoclet.
342    *
343    * @param sName
344    *        xdoclet name
345    * @param aAttributes
346    *        Attributes to be added
347    * @return Map with the key/value pairs
348    */
349   @Nonnull
350   public Map <String, String> addXdoclet (@Nonnull final String sName, @Nonnull final Map <String, String> aAttributes)
351   {
352     JCValueEnforcer.notNull (sName, "Name");
353     JCValueEnforcer.notNull (aAttributes, "Attributes");
354     final Map <String, String> p = addXdoclet (sName);
355     p.putAll (aAttributes);
356     return p;
357   }
358 
359   /**
360    * add an xdoclet with <code>@name attribute = "value"</code>. If value is
361    * <code>null</code> than it will be <code>@name attribute</code>.
362    *
363    * @param sName
364    *        xdoclet name
365    * @param sAttribute
366    *        Attribute name to be added
367    * @param sValue
368    *        Attribute value to be added
369    * @return Map with the key/value pairs
370    */
371   @Nonnull
372   public Map <String, String> addXdoclet (@Nonnull final String sName,
373                                           @Nonnull final String sAttribute,
374                                           @Nullable final String sValue)
375   {
376     JCValueEnforcer.notNull (sName, "Name");
377     JCValueEnforcer.notNull (sAttribute, "Attribute");
378     final Map <String, String> p = addXdoclet (sName);
379     p.put (sAttribute, sValue);
380     return p;
381   }

394   public void generate (@Nonnull final IJFormatter f)
395   {
396     // Is any "@" comment present?
397     final boolean bHasAt = !m_aAtParams.isEmpty () ||
398                            m_aAtReturn != null ||
399                            !m_aAtThrows.isEmpty () ||
400                            !m_aAtTags.isEmpty () ||
401                            !m_aAtXdoclets.isEmpty ();
402     if (!isEmpty () || bHasAt)
403     {
404       final boolean bIsJavaDoc = true;
405       final String sIndent = m_bIsSingleLineMode ? "// " : " * ";
406       final String sIndentLarge = sIndent + "    ";
407 
408       // Start comment
409       if (!m_bIsSingleLineMode)
410         f.print (bIsJavaDoc ? "/**" : "/*").newline ();
411 
412       // Print all simple text elements
413       format (f, sIndent);
414       if (!isEmpty () && bHasAt)
415         f.print (sIndent).newline ();
416 
417       for (final Map.Entry <String, JCommentPart> aEntry : m_aAtParams.entrySet ())
418       {
419         f.print (sIndent + "@param ").print (aEntry.getKey ()).newline ();
420         aEntry.getValue ().format (f, sIndentLarge);
421       }
422       if (m_aAtReturn != null)
423       {
424         f.print (sIndent + "@return").newline ();
425         m_aAtReturn.format (f, sIndentLarge);
426       }
427       for (final Map.Entry <AbstractJClass, JCommentPart> aEntry : m_aAtThrows.entrySet ())
428       {
429         f.print (sIndent + "@throws ").type (aEntry.getKey ()).newline ();
430         aEntry.getValue ().format (f, sIndentLarge);
431       }
432       for (final Map.Entry <String, JCommentPart> aEntry : m_aAtTags.entrySet ())
433       {
434         f.print (sIndent + "@" + aEntry.getKey () + " ");
435         aEntry.getValue ().format (f, "");
436       }
437       for (final Map.Entry <String, Map <String, String>> aEntry : m_aAtXdoclets.entrySet ())
438       {
439         f.print (sIndent + "@").print (aEntry.getKey ());
440         if (aEntry.getValue () != null)
441         {
442           for (final Map.Entry <String, String> aEntry2 : aEntry.getValue ().entrySet ())
443           {
444             final String sName = aEntry2.getKey ();
445             f.print (" ").print (sName);
446 
447             // Print value only if present
448             final String sValue = aEntry2.getValue ();
449             if (sValue != null && sValue.length () > 0)
450               f.print ("= \"").print (sValue).print ("\"");
451           }
452         }
453         f.newline ();
454       }
455 
456       // End comment
457       if (!m_bIsSingleLineMode)
458         f.print (" */").newline ();
459     }
