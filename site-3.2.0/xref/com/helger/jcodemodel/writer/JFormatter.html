91      /** true if this name is used as an identifier (like a variable name.) **/

94      public NameUsage (@Nonnull final String sName)
95      {
96        m_sName = sName;
97      }

99      /**
100      * @param aEnclosingClass
101      *        the class to check
102      * @return <code>true</code> if the short name is ambiguous in context of
103      *         enclosingClass and classes with this name can't be imported.
104      */
105     public boolean isAmbiguousIn (@Nonnull final JDefinedClass aEnclosingClass)
106     {
107       // more than one type with the same name
108       if (m_aReferencedClasses.size () > 1)
109         return true;
110 
111       // an id and (at least one) type with the same name
112       if (m_bIsVariableName && !m_aReferencedClasses.isEmpty ())
113         return true;
114 
115       // no references is always unambiguous
116       if (m_aReferencedClasses.isEmpty ())
117         return false;
118 
119       // we have exactly one reference
120       AbstractJClass aSingleRef = m_aReferencedClasses.get (0);
121       if (aSingleRef instanceof JAnonymousClass)
122       {
123         aSingleRef = ((JAnonymousClass) aSingleRef).base ();
124       }
125 
126       // special case where a generated type collides with a type in package
127       // java.lang
128       if (aSingleRef._package () == JFormatter.this.m_aPckJavaLang)
129       {
130         // make sure that there's no other class with this name within the
131         // same package
132         for (final JDefinedClass aClass : aEnclosingClass._package ().classes ())
133         {
134           // even if this is the only "String" class we use,
135           // if the class called "String" is in the same package,
136           // we still need to import it.
137           if (aClass.name ().equals (aSingleRef.name ()))
138           {
139             // collision -> ambiguous
140             return true;
141           }
142         }
143       }
144 
145       return false;
146     }

148     public boolean addReferencedType (@Nonnull final AbstractJClass aClazz)
149     {
150       if (false)

152       if (m_aReferencedClasses.contains (aClazz))
153         return false;
154       return m_aReferencedClasses.add (aClazz);
155     }

175     public void setVariableName ()
176     {
177       // Check if something can be a variable or a type
178       for (final AbstractJClass aRefedType : m_aReferencedClasses)
179       {
180         if (aRefedType.outer () != null)
181         {
182           m_bIsVariableName = false;
183           return;
184         }
185       }
186       m_bIsVariableName = true;
187     }

246     @Nullable
247     private AbstractJClassbstractJClass _getClassForImport (@Nullable final AbstractJClass aClass)
248     {
249       AbstractJClass aRealClass = aClass;
250       if (aRealClass instanceof JAnonymousClass)
251       {
252         // get the super class of the anonymous class
253         return _getClassForImport (((JAnonymousClass) aRealClass).base ());
254       }
255       if (aRealClass instanceof JNarrowedClass)
256       {
257         // Never imported narrowed class but the erasure only
258         aRealClass = aRealClass.erasure ();
259       }
260       return aRealClass;
261     }
262 
263     public void addDontImportClass (@Nonnull final AbstractJClass aClass)
264     {
265       final AbstractJClass aRealClass = _getClassForImport (aClass);
266       m_aDontImportClasses.add (aRealClass);
267     }
268 
269     public boolean add (@Nonnull final AbstractJClass aClass)
270     {
271       final AbstractJClass aRealClass = _getClassForImport (aClass);
272 
273       if (m_aDontImportClasses.contains (aRealClass))
274       {
275         if (m_bDebugImport)
276           System.out.println ("The class '" + aRealClass.fullName () + "' should not be imported!");
277         return false;
278       }
279 
280       // Avoid importing 2 classes with the same class name
281       if (!m_aNames.add (aRealClass.name ()))
282       {
283         if (m_bDebugImport)
284           System.out.println ("A class with local name '" + aRealClass.name () + "' is already in the import list.");
285         return false;
286       }
287 
288       if (!m_aClasses.add (aRealClass))
289       {
290         if (m_bDebugImport)
291           System.out.println ("The class '" + aRealClass.fullName () + "' is already in the import list.");
292         return false;
293       }
294 
295       if (m_bDebugImport)
296         System.out.println ("Added import class '" + aClass.fullName () + "'");
297       return true;
298     }

302       final AbstractJClass aRealClass = _getClassForImport (aClass);

304       return m_aClasses.contains (aRealClass);

425   private static boolean _needSpace (final char c1, final char c2)
426   {
427     if ((c1 == ']') && (c2 == '{'))
428       return true;
429     if (c1 == ';')
430       return true;
431     if (c1 == CLOSE_TYPE_ARGS)
432     {
433       // e.g., "public Foo<Bar> test;"
434       if (c2 == '(')
435       {
436         // but not "new Foo<Bar>()"
437         return false;
438       }
439       return true;
440     }
441     if ((c1 == ')') && (c2 == '{'))
442       return true;
443     if ((c1 == ',') || (c1 == '='))
444       return true;
445     if (c2 == '=')
446       return true;
447     if (Character.isDigit (c1))
448     {
449       if ((c2 == '(') || (c2 == ')') || (c2 == ';') || (c2 == ','))
450         return false;
451       return true;
452     }
453     if (Character.isJavaIdentifierPart (c1))
454     {
455       switch (c2)
456       {
457         case '{':
458         case '}':
459         case '+':
460         case '-':
461         case '>':
462         case '@':
463           return true;
464         default:
465           return Character.isJavaIdentifierStart (c2);
466       }
467     }
468     if (Character.isJavaIdentifierStart (c2))
469     {
470       switch (c1)
471       {
472         case ']':
473         case ')':
474         case '}':
475         case '+':
476           return true;
477         default:
478           return false;
479       }
480     }
481     if (Character.isDigit (c2))
482     {
483       if (c1 == '(')
484         return false;
485       return true;
486     }
487     return false;
488   }

490   private void _spaceIfNeeded (final char c)
491   {
492     if (m_bAtBeginningOfLine)
493     {
494       for (int i = 0; i < m_nIndentLevel; i++)
495         m_aPW.print (m_sIndentString);
496       m_bAtBeginningOfLine = false;
497     }
498     else
499       if (m_cLastChar != 0 && _needSpace (m_cLastChar, c))
500         m_aPW.print (' ');
501   }

503   @Nonnull
504   public JFormatter print (final char c)
505   {
506     if (m_eMode == EMode.PRINTING)
507     {
508       if (c == CLOSE_TYPE_ARGS)
509       {
510         m_aPW.print ('>');
511       }
512       else
513       {
514         _spaceIfNeeded (c);
515         m_aPW.print (c);
516       }
517       m_cLastChar = c;
518     }
519     return this;
520   }

522   @Nonnull
523   public JFormatter print (@Nonnull final String sStr)
524   {
525     if (m_eMode == EMode.PRINTING && sStr.length () > 0)
526     {
527       _spaceIfNeeded (sStr.charAt (0));
528       m_aPW.print (sStr);
529       m_cLastChar = sStr.charAt (sStr.length () - 1);
530     }
531     return this;
532   }

534   @Nonnull
535   public JFormatter type (@Nonnull final AbstractJClass aType)
536   {
537     switch (m_eMode)
538     {
539       case COLLECTING:
540         if (!aType.isError ())
541         {
542           final String sShortName = aType.name ();
543           NameUsage aUsages = m_aCollectedReferences.get (sShortName);
544           if (aUsages == null)
545           {
546             aUsages = new NameUsage (sShortName);
547             m_aCollectedReferences.put (sShortName, aUsages);
548           }
549           aUsages.addReferencedType (aType);
550         }
551         break;
552       case PRINTING:
553         if (aType.isError ())
554         {
555           print ("Object");
556         }
557         else
558           // many of the JTypes in this list are either primitive or belong to
559           // package java so we don't need a FQCN
560           if (m_aImportedClasses.contains (aType) || aType._package () == m_aPckJavaLang)
561           {
562             // FQCN imported or not necessary, so generate short name
563             print (aType.name ());
564           }
565           else
566           {
567             final AbstractJClass aOuter = aType.outer ();
568             if (aOuter != null)
569             {
570               type (aOuter).print ('.').print (aType.name ());
571             }
572             else
573             {
574               // collision was detected, so generate FQCN
575               print (aType.fullName ());
576             }
577           }
578         break;
579       case FIND_ERROR_TYPES:
580         if (aType.isError ())
581           m_bContainsErrorTypes = true;
582         break;
583     }
584     return this;
585   }

587   @Nonnull
588   public JFormatter id (@Nonnull final String sID)
589   {
590     switch (m_eMode)
591     {
592       case COLLECTING:
593         // see if there is a type name that collides with this id
594         NameUsage aUsages = m_aCollectedReferences.get (sID);
595         if (aUsages == null)
596         {
597           // not a type, but we need to create a place holder to
598           // see if there might be a collision with a type
599           aUsages = new NameUsage (sID);
600           m_aCollectedReferences.put (sID, aUsages);
601         }
602         aUsages.setVariableName ();
603         break;
604       case PRINTING:
605         print (sID);
606         break;
607     }
608     return this;
609   }

668   private boolean _collectCausesNoAmbiguities (@Nonnull final AbstractJClass aReference,
669                                                @Nonnull final JDefinedClass aClassToBeWritten)
670   {
671     if (m_bDebugImport)
672       System.out.println ("_collectCausesNoAmbiguities(" +
673                           aReference.fullName () +
674                           ", " +
675                           aClassToBeWritten.fullName () +
676                           ")");
677 
678     final NameUsage aUsages = m_aCollectedReferences.get (aReference.name ());
679     if (aUsages == null)
680       return true;
681     return !aUsages.isAmbiguousIn (aClassToBeWritten) && aUsages.containsReferencedType (aReference);
682   }

684   /**
685    * determine if an import statement should be used for given class. This is a
686    * matter of style and convention
687    *
688    * @param aReference
689    *        {@link AbstractJClass} referenced class
690    * @param aClassToBeWritten
691    *        {@link AbstractJClass} currently generated class
692    * @return <code>true</code> if an import statement can be used to shorten
693    *         references to referenced class
694    */
695   private boolean _collectShouldBeImported (@Nonnull final AbstractJClass aReference,
696                                             @Nonnull final JDefinedClass aClassToBeWritten)
697   {
698     if (m_bDebugImport)
699       System.out.println ("_collectShouldBeImported(" +
700                           aReference.fullName () +
701                           ", " +
702                           aClassToBeWritten.fullName () +
703                           ")");
704 
705     AbstractJClass aRealReference = aReference;
706     if (aRealReference instanceof JAnonymousClass)
707     {
708       // get the super class of the anonymous class
709       aRealReference = ((JAnonymousClass) aRealReference).base ();
710     }
711     if (aRealReference instanceof JNarrowedClass)
712     {
713       // Remove the generic arguments
714       aRealReference = aRealReference.erasure ();
715     }
716 
717     // Is it an inner class?
718     final AbstractJClass aOuter = aRealReference.outer ();
719     if (aOuter != null)
720     {
721       // Import inner class only when it's name contain a name of enclosing
722       // class.
723       // In such case no information is lost when we refer to inner class
724       // without mentioning it's enclosing class
725       if (aRealReference.name ().contains (aOuter.name ()))
726       {
727         // Recurse
728         if (_collectShouldBeImported (aOuter, aClassToBeWritten))
729           return true;
730       }
731 
732       // Do not import inner classes in all other cases to aid
733       // understandability/readability.
734       return false;
735     }
736     return true;
737   }

739   /**
740    * If reference is inner-class adds some outer class to the list of imported
741    * classes if it
742    *
743    * @param aClassToBeWritten
744    *        {@link AbstractJClass} that may or may not have an import
745    * @param aGeneratingClass
746    *        {@link AbstractJClass} that is the current class being processed
747    */
748   private void _collectImportOuterClassIfCausesNoAmbiguities (@Nonnull final AbstractJClass aReference,
749                                                               @Nonnull final JDefinedClass aClassToBeWritten)
750   {
751     if (m_bDebugImport)
752       System.out.println ("_collectImportOuterClassIfCausesNoAmbiguities(" +
753                           aReference.fullName () +
754                           ", " +
755                           aClassToBeWritten.fullName () +
756                           ")");
757 
758     final AbstractJClass aOuter = aReference.outer ();
759     if (aOuter != null)
760     {
761       if (_collectCausesNoAmbiguities (aOuter, aClassToBeWritten) &&
762           _collectShouldBeImported (aOuter, aClassToBeWritten))
763       {
764         m_aImportedClasses.add (aOuter);
765       }
766       else
767       {
768         // Recursive call
769         _collectImportOuterClassIfCausesNoAmbiguities (aOuter, aClassToBeWritten);
770       }
771     }
772   }

779    * @param aGeneratingClass
780    *        {@link AbstractJClass} that is the current class being processed
781    * @return true if an import statement should be suppressed, false otherwise

783   private boolean _printIsImplicitlyImported (@Nonnull final AbstractJClass aReference,
784                                               @Nonnull final AbstractJClass aClassToBeWrittem)
785   {
786     if (m_bDebugImport)
787       System.out.println ("_printIsImplicitlyImported(" +
788                           aReference.fullName () +
789                           ", " +
790                           aClassToBeWrittem.fullName () +
791                           ")");
792 
793     AbstractJClass aRealReference = aReference;
794     if (aRealReference instanceof JAnonymousClass)
795     {
796       // Get the super class of the anonymous class
797       aRealReference = ((JAnonymousClass) aRealReference).base ();
798     }
799     if (aRealReference instanceof JNarrowedClass)
800     {
801       // Remove generic type arguments
802       aRealReference = aRealReference.erasure ();
803     }
804 
805     final JPackage aPackage = aRealReference._package ();
806     if (aPackage == null)
807     {
808       // May be null for JTypeVar and JTypeWildcard
809       return true;
810     }
811 
812     if (aPackage.isUnnamed ())
813     {
814       // Root package - no need to import something
815       return true;
816     }
817 
818     if (aPackage == m_aPckJavaLang)
819     {
820       // no need to explicitly import java.lang classes
821       return true;
822     }
823 
824     // All pkg local classes do not need an
825     // import stmt for ref, except for inner classes
826     if (aPackage == aClassToBeWrittem._package ())
827     {
828       AbstractJClass aOuter = aRealReference.outer ();
829       if (aOuter == null) // top-level class
830       {
831         // top-level package-local class needs no explicit import
832         return true;
833       }
834 
835       // inner-class
836       AbstractJClass aTopLevelClass = aOuter;
837       aOuter = aTopLevelClass.outer ();
838       while (aOuter != null)
839       {
840         aTopLevelClass = aOuter;
841         aOuter = aTopLevelClass.outer ();
842       }
843 
844       // if reference is inner-class and
845       // reference's top-level class is generated clazz,
846       // i. e. reference is enclosed in generated clazz,
847       // then it needs no explicit import statement.
848       return aTopLevelClass == aClassToBeWrittem;
849     }
850     return false;
851   }
