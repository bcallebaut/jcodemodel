41  package com.helger.jcodemodel;
42  
43  import java.io.Closeable;
44  import java.util.Collection;
45  
46  import javax.annotation.Nonnull;
47  
48  /**
49   * Base interface for JFormatter.
50   *
51   * @author Philip Helger
52   */
53  public interface IJFormatter extends Closeable
54  {
55    /**
56     * Special character token we use to differentiate '&gt;' as an operator and
57     * '&gt;' as the end of the type arguments. The former uses '&gt;' and it
58     * requires a preceding whitespace. The latter uses this, and it does not have
59     * a preceding whitespace.
60     */
61    char CLOSE_TYPE_ARGS = '\uFFFF';
62  
63    /**
64     * @return <code>true</code> if we are in the printing mode, where we actually
65     *         produce text. The other (internal) mode is the "collecting mode".
66     */
67    boolean isPrinting ();
68  
69    /**
70     * Increment the indentation level.
71     *
72     * @return this for chaining
73     */
74    @Nonnull
75    IJFormatter indent ();
76  
77    /**
78     * Decrement the indentation level.
79     *
80     * @return this for chaining
81     */
82    @Nonnull
83    IJFormatter outdent ();
84  
85    /**
86     * Print a new line into the stream
87     *
88     * @return this for chaining
89     */
90    @Nonnull
91    IJFormatter newline ();
92  
93    /**
94     * Print a char into the stream
95     *
96     * @param c
97     *        the char
98     * @return this for chaining
99     */
100   @Nonnull
101   IJFormatter print (char c);
102 
103   @Nonnull
104   default IJFormatter printCloseTypeArgs ()
105   {
106     return print (CLOSE_TYPE_ARGS);
107   }
108 
109   /**
110    * Print a String into the stream. Indentation happens automatically.
111    *
112    * @param sStr
113    *        the String
114    * @return this
115    */
116   @Nonnull
117   IJFormatter print (@Nonnull String sStr);
118 
119   /**
120    * Print a type name.
121    * <p>
122    * In the collecting mode we use this information to decide what types to
123    * import and what not to.
124    *
125    * @param aType
126    *        Type to be emitted
127    * @return this for chaining
128    */
129   @Nonnull
130   IJFormatter type (@Nonnull AbstractJClass aType);
131 
132   @Nonnull
133   default IJFormatter type (@Nonnull final AbstractJType aType)
134   {
135     if (aType.isReference ())
136       return type ((AbstractJClass) aType);
137     return generable (aType);
138   }
139 
140   /**
141    * Cause the {@link JVar} to generate source for itself. With annotations,
142    * type, name and init expression.
143    *
144    * @param aVar
145    *        the {@link JVar} object
146    * @return this for chaining
147    */
148   @Nonnull
149   IJFormatter var (@Nonnull JVar aVar);
150 
151   /**
152    * Print an identifier
153    *
154    * @param sID
155    *        identifier
156    * @return this for chaining
157    */
158   @Nonnull
159   IJFormatter id (@Nonnull String sID);
160 
161   /**
162    * Cause the {@link IJGenerable} object to generate source for itself
163    *
164    * @param aObj
165    *        the object
166    * @return this for chaining
167    */
168   @Nonnull
169   IJFormatter generable (@Nonnull IJGenerable aObj);
170 
171   /**
172    * Produces {@link IJGenerable}s separated by ','
173    *
174    * @param aList
175    *        List of {@link IJGenerable} objects that will be separated by a
176    *        comma
177    * @return this for chaining
178    */
179   @Nonnull
180   IJFormatter generable (@Nonnull final Collection <? extends IJGenerable> aList);
181 
182   /**
183    * Cause the {@link IJStatement} to generate source for itself
184    *
185    * @param aObj
186    *        the object
187    * @return this for chaining
188    */
189   @Nonnull
190   IJFormatter statement (@Nonnull IJStatement aObj);
191 
192   /**
193    * Cause the {@link IJDeclaration} to generate source for itself
194    *
195    * @param aObj
196    *        the object
197    * @return this for chaining
198    */
199   @Nonnull
200   IJFormatter declaration (@Nonnull IJDeclaration aObj);
201 }
