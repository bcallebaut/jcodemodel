171                                @Nonnull final String sName,
172                                @Nonnull final EClassType eClassType) throws JClassAlreadyExistsException
173   {
174     if (m_aClasses.containsKey (sName))
175       throw new JClassAlreadyExistsException (m_aClasses.get (sName));
176 
177     // XXX problems caught in the NC constructor
178     final JDefinedClassss.html#JDefinedClass">JDefinedClass c = new JDefinedClass (this, nMods, sName, eClassType);
179 
180     if (m_aUpperCaseClassMap != null)
181     {
182       final String sUpperName = sName.toUpperCase ();
183       final JDefinedClass dc = m_aUpperCaseClassMap.get (sUpperName);
184       if (dc != null)
185         throw new JClassAlreadyExistsException (dc);
186       m_aUpperCaseClassMap.put (sUpperName, c);
187     }
188     m_aClasses.put (sName, c);

291   /**
292    * Removes a class from this package.
293    *
294    * @param aClass
295    *        Class to be removed. May not be <code>null</code>.
296    */
297   public void remove (@Nonnull final AbstractJClass aClass)
298   {
299     JCValueEnforcer.isTrue (aClass._package () == this,
300                             () -> "the specified class (" +
301                                   aClass.fullName () +
302                                   ") is not a member of this package (" +
303                                   name () +
304                                   "), or it is a referenced class");
305 
306     // note that c may not be a member of classes.
307     // this happens when someone is trying to remove a non generated class
308     m_aClasses.remove (aClass.name ());
309     if (m_aUpperCaseClassMap != null)
310       m_aUpperCaseClassMap.remove (aClass.name ().toUpperCase ());
311   }

313   /**
314    * Reference a class within this package.
315    *
316    * @param sClassLocalName
317    *        Local class name to reference
318    * @return The referenced class
319    * @throws ClassNotFoundException
320    *         If the provided class does not exist
321    */
322   @Nonnull
323   public AbstractJClass ref (@Nonnull final String sClassLocalName) throws ClassNotFoundException
324   {
325     JCValueEnforcer.isTrue (sClassLocalName.indexOf ('.') < 0, () -> "JClass name contains '.': " + sClassLocalName);
326 
327     String sFQCN;
328     if (isUnnamed ())
329       sFQCN = "";
330     else
331       sFQCN = m_sName + '.';
332     sFQCN += sClassLocalName;
333 
334     return m_aOwner.ref (Class.forName (sFQCN));
335   }

457   boolean buildsErrorTypeRefs ()
458   {
459     // check classes
460     for (final JDefinedClass c : m_aClasses.values ())
461     {
462       if (c.isHidden ())
463       {
464         // don't check this file
465         continue;
466       }
467 
468       if (c.containsErrorTypes ())
469         return true;
470     }
471     return false;
472   }

474   /* package */int countArtifacts ()
475   {
476     int ret = 0;
477     for (final JDefinedClass c : m_aClasses.values ())
478     {
479       if (c.isHidden ())
480       {
481         // don't generate this file
482         continue;
483       }
484       ret++;
485     }
486 
487     if (m_aAnnotations != null || m_aJavaDoc != null)
488     {
489       // package-info
490       ret++;
