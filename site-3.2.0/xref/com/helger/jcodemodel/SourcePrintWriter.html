41  package com.helger.jcodemodel;
42  
43  import java.io.FilterWriter;
44  import java.io.IOException;
45  import java.io.Writer;
46  
47  import javax.annotation.Nonnull;
48  
49  /**
50   * A special version of {@link java.io.PrintWriter} that has a customizable new
51   * line string.
52   *
53   * @author Philip Helger
54   */
55  public final class SourcePrintWriter extends FilterWriter
56  {
57    private final String m_sNewLine;
58  
59    public SourcePrintWriter (@Nonnull final Writer aWrappedWriter, @Nonnull final String sNewLine)
60    {
61      super (aWrappedWriter);
62      m_sNewLine = sNewLine;
63    }
64  
65    private void _handleException (@Nonnull final IOException ex, @Nonnull final String sSource)
66    {
67      System.err.println ("Error on Writer: " + sSource);
68      ex.printStackTrace ();
69    }
70  
71    private void _write (final char c)
72    {
73      try
74      {
75        super.write (c);
76      }
77      catch (final IOException ex)
78      {
79        _handleException (ex, "write char");
80      }
81    }
82  
83    private void _write (@Nonnull final String sStr)
84    {
85      try
86      {
87        super.write (sStr, 0, sStr.length ());
88      }
89      catch (final IOException ex)
90      {
91        _handleException (ex, "write String");
92      }
93    }
94  
95    public void print (final char c)
96    {
97      _write (c);
98    }
99  
100   public void print (@Nonnull final String sStr)
101   {
102     _write (sStr);
103   }
104 
105   public void println ()
106   {
107     _write (m_sNewLine);
108   }
109 
110   public void println (final String sStr)
111   {
112     _write (sStr);
113     _write (m_sNewLine);
114   }
115 
116   @Override
117   public void close ()
118   {
119     try
120     {
121       super.close ();
122     }
123     catch (final IOException ex)
124     {
125       _handleException (ex, "close");
126     }
127   }
128 }
