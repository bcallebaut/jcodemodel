49  /**
50   * Simple program to test the generation of the varargs feature in jdk 1.5
51   *
52   * @author Bhakti Mehta Bhakti.Mehta@sun.com
53   */
54  /*
55   * ====================================================== This is how the output
56   * from this program looks like Still need to learn how to work on instantiation
57   * and args ========================================================= public
58   * class Test { public void foo(java.lang.String param1, java.lang.Integer
59   * param2, java.lang.String param5, java.lang.Object... param3) { for (int count
60   * = 0; (count<(param3.length)); count ++) {
61   * java.lang.System.out.println((param3[count])); } } public static void
62   * main(java.lang.String[] args) { } }
63   * ==========================================================
64   */
65  
66  public final class VarArgsFuncTest
67  {
68  
69    @Test
70    public void testBasic () throws Exception
71    {
72      final JCodeModel cm = new JCodeModel ();
73      final JDefinedClass cls = cm._class ("Test");
74      final JMethod m = cls.method (JMod.PUBLIC, cm.VOID, "foo");
75      m.param (String.class, "param1");
76      m.param (Integer.class, "param2");
77      final JVar var = m.varParam (Object.class, "param3");
78  
79      if (false)
80        System.out.println ("First varParam " + var);
81  
82      // checking for param after varParam it behaves ok
83      // JVar[] var1 = m.varParam(Float.class, "param4");
84      final AbstractJClass string = cm.ref (String.class);
85      final AbstractJClass stringArray = string.array ();
86      // JVar param5 =
87      m.param (String.class, "param5");
88  
89      final JForLoop forloop = m.body ()._for ();
90  
91      final JVar jcount = forloop.init (cm.INT, "count", JExpr.lit (0));
92  
93      forloop.test (jcount.lt (JExpr.direct ("param3.length")));
94      forloop.update (jcount.incr ());
95  
96      final JFieldRef out = cm.ref (System.class).staticRef ("out");
97  
98      final JVar typearray = m.varParam ();
99      assertNotNull (typearray);
100 
101     // JInvocation invocation =
102     forloop.body ().invoke (out, "println").arg (JExpr.direct ("param3[count]"));
103 
104     final JMethod main = cls.method (JMod.PUBLIC | JMod.STATIC, cm.VOID, "main");
105     main.param (stringArray, "args");
106     main.body ()
107         .directStatement ("new Test().foo(new String(\"Param1\"),new Integer(5),null,new String(\"Param3\"),new String(\"Param4\"));");
108 
109     CodeModelTestsHelper.parseCodeModel (cm);
110   }
