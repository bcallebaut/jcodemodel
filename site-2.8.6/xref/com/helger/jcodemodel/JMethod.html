116   /**
117    * Constructor for regular methods
118    *
119    * @param aOwningClass
120    *        Outer class. May not be <code>null</code>.
121    * @param mods
122    *        Modifiers for this method's declaration
123    * @param aReturnType
124    *        Return type for the method. May not be <code>null</code>.
125    * @param sName
126    *        Name of this method. May neither be <code>null</code> nor empty.
127    */
128   protected JMethod (@Nonnull final JDefinedClass aOwningClass,
129                      final int mods,
130                      @Nonnull final AbstractJType aReturnType,
131                      @Nonnull final String sName)
132   {
133     JCValueEnforcer.notNull (aOwningClass, "OwningClass");
134     JCValueEnforcer.notNull (aReturnType, "ReturnType");
135     JCValueEnforcer.notEmpty (sName, "Name");
136     m_aMods = JMods.forMethod (mods);
137     m_aReturnType = aReturnType;
138     m_sName = sName;
139     m_aOwningClass = aOwningClass;
140   }

310    * Add the specified variable argument to the list of parameters for this
311    * method signature.
312    *
313    * @param mods
314    *        mods to use
315    * @param type
316    *        Type of the parameter being added. Is automatically converted to an
317    *        array.
318    * @param name
319    *        Name of the parameter being added
320    * @return the created variable parameter
321    * @throws IllegalStateException
322    *         If this method is called twice. varargs in J2SE 1.5 can appear only
323    *         once in the method signature.
324    */
325   @Nonnull
326   public JVar varParam (final int mods, @Nonnull final AbstractJType type, @Nonnull final String name)
327   {
328     if (hasVarArgs ())
329       throw new IllegalStateException ("Cannot have two varargs in a method,\n" +
330                                        "Check if varParam method of JMethod is" +
331                                        " invoked more than once");
332 
333     m_aVarParam = new JVar (JMods.forVar (mods), type.array (), name, null);
334     return m_aVarParam;
335   }

482   /**
483    * Returns true if the method has the specified signature.
484    *
485    * @param argTypes
486    *        Signature to check
487    * @return <code>true</code> if this method has the provided signature
488    */
489   public boolean hasSignature (@Nonnull final AbstractJType [] argTypes)
490   {
491     final JVar [] aParams = listParams ();
492     if (aParams.length != argTypes.length)
493       return false;
494 
495     for (int i = 0; i < aParams.length; i++)
496       if (!aParams[i].type ().equals (argTypes[i]))
497         return false;
498 
499     return true;
500   }

534   @Override
535   public void declare (@Nonnull final JFormatter f)
536   {
537     if (m_aJDoc != null)
538       f.generable (m_aJDoc);
539 
540     if (m_aAnnotations != null)
541       for (final JAnnotationUse a : m_aAnnotations)
542         f.generable (a).newline ();
543 
544     f.generable (m_aMods);
545 
546     // declare the generics parameters
547     super.declare (f);
548 
549     if (!isConstructor ())
550       f.generable (m_aReturnType);
551     f.id (m_sName).print ('(').indent ();
552     // when parameters are printed in new lines, we want them to be indented.
553     // there's a good chance no newlines happen, too, but just in case it does.
554     boolean first = true;
555     for (final JVar var : m_aParams)
556     {
557       if (!first)
558         f.print (',');
559       if (var.isAnnotated ())
560         f.newline ();
561       f.var (var);
562       first = false;
563     }
564     if (hasVarArgs ())
565     {
566       if (!first)
567         f.print (',');
568       for (final JAnnotationUse annotation : m_aVarParam.annotations ())
569         f.generable (annotation).newline ();
570       f.generable (m_aVarParam.mods ()).generable (m_aVarParam.type ().elementType ());
571       f.print ("... ");
572       f.id (m_aVarParam.name ());
573     }
574 
575     f.outdent ().print (')');
576     if (m_aThrows != null && !m_aThrows.isEmpty ())
577     {
578       f.newline ().indent ().print ("throws").generable (m_aThrows).newline ().outdent ();
579     }
580 
581     if (m_aDefaultValue != null)
582     {
583       // For annotation values
584       f.print ("default ");
585       f.generable (m_aDefaultValue);
586     }
587     if (m_aBody != null)
588     {
589       f.statement (m_aBody);
590     }
591     else
592     {
593       final boolean bIsDeclarationOnly = (m_aOwningClass.isInterface () && !m_aMods.isDefault ()) ||
594                                          m_aOwningClass.isAnnotationTypeDeclaration () ||
595                                          m_aMods.isAbstract () ||
596                                          m_aMods.isNative ();
597 
598       if (bIsDeclarationOnly)
599       {
600         f.print (';').newline ();
601       }
602       else
603       {
604         // Print an empty body for non-native, non-abstract methods
605         f.statement (new JBlock ());
606       }
607     }
608   }
