41  package com.helger.jcodemodel.writer;
42  
43  import java.io.FilterOutputStream;
44  import java.io.IOException;
45  import java.io.OutputStream;
46  import java.util.zip.ZipEntry;
47  import java.util.zip.ZipOutputStream;
48  
49  import javax.annotation.Nonnull;
50  
51  import com.helger.jcodemodel.AbstractCodeWriter;
52  import com.helger.jcodemodel.JPackage;
53  
54  /**
55   * Writes all the files into a zip file.
56   *
57   * @author Kohsuke Kawaguchi (kohsuke.kawaguchi@sun.com)
58   */
59  public class ZipCodeWriter extends AbstractCodeWriter
60  {
61    private final ZipOutputStream m_aZOS;
62    private final OutputStream m_aFOS;
63  
64    /**
65     * @param target
66     *        Zip file will be written to this stream.
67     */
68    public ZipCodeWriter (@Nonnull final OutputStream target)
69    {
70      this (target, getDefaultNewLine ());
71    }
72  
73    /**
74     * @param target
75     *        Zip file will be written to this stream.
76     * @param sNewLine
77     *        The new line string to be used for source files
78     */
79    public ZipCodeWriter (@Nonnull final OutputStream target, @Nonnull final String sNewLine)
80    {
81      super (null, sNewLine);
82      m_aZOS = new ZipOutputStream (target);
83      // nullify the close method.
84      m_aFOS = new FilterOutputStream (m_aZOS)
85      {
86        @Override
87        public void close ()
88        {}
89      };
90    }
91  
92    @Override
93    public OutputStream openBinary (@Nonnull final JPackage pkg, @Nonnull final String fileName) throws IOException
94    {
95      String name = fileName;
96      if (!pkg.isUnnamed ())
97        name = _toDirName (pkg) + name;
98  
99      m_aZOS.putNextEntry (new ZipEntry (name));
100     return m_aFOS;
101   }
102 
103   /** Converts a package name to the directory name. */
104   private static String _toDirName (@Nonnull final JPackage pkg)
105   {
106     return pkg.name ().replace ('.', '/') + '/';
107   }
108 
109   @Override
110   public void close () throws IOException
111   {
112     m_aZOS.close ();
113   }
114 }
