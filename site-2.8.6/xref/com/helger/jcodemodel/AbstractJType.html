48  /**
49   * A representation of a type in codeModel. A type is always either primitive (
50   * {@link JPrimitiveType}) or a reference type ({@link AbstractJClass}).<br>
51   * Note: up to version 2.7.6 this class implemented
52   * <code>Comparable &lt;AbstractJType&gt;</code>. Since this was specific to
53   * import handling on emitting code, it was removed with 2.7.7!
54   */
55  public abstract class AbstractJType implements IJGenerable, IJOwned
56  {
57    /**
58     * Obtains a reference to the primitive type object from a type name.
59     *
60     * @param codeModel
61     *        Base code model
62     * @param sTypeName
63     *        primitive type to be parsed (e.g. "int" or "void")
64     * @return Never <code>null</code>
65     * @throws IllegalArgumentException
66     *         If the passed type name is not a primitive type name
67     */
68    @Nonnull
69    public static JPrimitiveType parse (@Nonnull final JCodeModel codeModel, @Nonnull final String sTypeName)
70    {
71      if (sTypeName.equals ("void"))
72        return codeModel.VOID;
73      if (sTypeName.equals ("boolean"))
74        return codeModel.BOOLEAN;
75      if (sTypeName.equals ("byte"))
76        return codeModel.BYTE;
77      if (sTypeName.equals ("short"))
78        return codeModel.SHORT;
79      if (sTypeName.equals ("char"))
80        return codeModel.CHAR;
81      if (sTypeName.equals ("int"))
82        return codeModel.INT;
83      if (sTypeName.equals ("float"))
84        return codeModel.FLOAT;
85      if (sTypeName.equals ("long"))
86        return codeModel.LONG;
87      if (sTypeName.equals ("double"))
88        return codeModel.DOUBLE;
89      throw new IllegalArgumentException ("Not a primitive type: " + sTypeName);
90    }

92    /**
93     * Gets the full name of the type. See
94     * http://java.sun.com/docs/books/jls/second_edition/html/names.doc.html#25430
95     * for the details.
96     *
97     * @return Strings like "int", "java.lang.String", "java.io.File[]". May be
98     *         <code>null</code> for unnamed classes.

251   /**
252    * Checks the relationship between two types.
253    * <p>
254    * This method performs superset of actions that are performed by
255    * {@link Class#isAssignableFrom(Class)} For example,
256    * baseClass.isAssignableFrom(derivedClass) is always <code>true</code>.
257    * <p>
258    * There are two differences of this method and
259    * {@link Class#isAssignableFrom(Class)}
260    * <ol>
261    * <li>This method works with primitive types
262    * <li>This method processes generic arguments and supports wildcards
263    * </ol>
264    * <p>
265    * Examples:
266    * <ol>
267    * <li>[[List]].isAssignableFrom ([[List&lt;T&gt;]])</li>
268    * <li>[[List&lt;T&gt;]].isAssignableFrom ([[List]])</li>
269    * <li>[[List&lt;? extends Object&gt;]].isAssignableFrom
270    * ([[List&lt;Integer&gt;]])</li>
271    * <li>[[List&lt;? super Serializable&gt;]].isAssignableFrom
272    * ([[List&lt;String&gt;]])</li>
273    * <li>[[List&lt;? super Serializable&gt;]].isAssignableFrom
274    * ([[List&lt;String&gt;]])</li>
275    * <li>[[List&lt;? extends Object&gt;]].isAssignableFrom ([[List&lt;? extends
276    * Integer&gt;]])</li>
277    * <li>[[List&lt;? extends List&lt;? extends Object&gt;&gt;]].isAssignableFrom
278    * ([[List&lt;List&lt;Integer&gt;&gt;]])</li>
279    * </ol>
280    *
281    * @param that
282    *        Type to check
283    * @return <code>true</code> if assignable, <code>false</code> if not
284    */
285   public boolean isAssignableFrom (@Nonnull final AbstractJType that)
286   {
287     return isAssignableFrom (that, true);
288   }
289 
290   protected boolean isAssignableFrom (@Nonnull final AbstractJType that, final boolean bAllowsRawTypeUnchekedConversion)
291   {
292     if (isError () || that.isError ())
293       return false;
294     if (this.equals (that))
295       return true;
296 
297     if (this.isReference () && that.isReference ())
298     {
299       final AbstractJClass thisClass = (AbstractJClass) this;
300       final AbstractJClass thatClass = (AbstractJClass) that;
301 
302       // Bottom: Anything anything = null
303       if (thatClass instanceof JNullType)
304         return true;
305 
306       // Top: Object object = (Anything)anything
307       if (thisClass == thisClass._package ().owner ().ref (Object.class))
308         return true;
309 
310       // Array covariance: i. e. Object[] array1 = (Integer[])array2
311       if (this.isArray () && that.isArray ())
312         return this.elementType ().isAssignableFrom (that.elementType (), false);
313 
314       if (thisClass.erasure ().equals (thatClass.erasure ()))
315       {
316         // Raw classes: i. e. List list1 = (List<T>)list2;
317         if (!thisClass.isParameterized ())
318           return true;
319 
320         // Raw classes unchecked conversion: i. e. List<T> list1 = (List)list2
321         if (!thatClass.isParameterized ())
322           return bAllowsRawTypeUnchekedConversion;
323 
324         for (int i = 0; i < thisClass.getTypeParameters ().size (); i++)
325         {
326           final AbstractJClass thisParameter = thisClass.getTypeParameters ().get (i);
327           final AbstractJClass thatParameter = thatClass.getTypeParameters ().get (i);
328 
329           if (thisParameter instanceof JTypeWildcard)
330           {
331             final JTypeWildcard thisWildcard = (JTypeWildcard) thisParameter;
332 
333             if (thatParameter instanceof JTypeWildcard)
334             {
335               final JTypeWildcard thatWildcard = (JTypeWildcard) thatParameter;
336               if (thisWildcard.boundMode () != thatWildcard.boundMode ())
337                 return false;
338               if (thisWildcard.boundMode () == JTypeWildcard.EBoundMode.EXTENDS)
339                 return thisWildcard.bound ().isAssignableFrom (thatWildcard.bound (), false);
340               if (thisWildcard.boundMode () == JTypeWildcard.EBoundMode.SUPER)
341                 return thatWildcard.bound ().isAssignableFrom (thisWildcard.bound (), false);
342               throw new IllegalStateException ("Unsupported wildcard bound mode: " + thisWildcard.boundMode ());
343             }
344 
345             if (thisWildcard.boundMode () == JTypeWildcard.EBoundMode.EXTENDS)
346               return thisWildcard.bound ().isAssignableFrom (thatParameter, false);
347             if (thisWildcard.boundMode () == JTypeWildcard.EBoundMode.SUPER)
348               return thatParameter.isAssignableFrom (thisWildcard.bound (), false);
349             throw new IllegalStateException ("Unsupported wildcard bound mode: " + thisWildcard.boundMode ());
350           }
351 
352           if (!thisParameter.equals (thatParameter))
353             return false;
354         }
355         return true;
356       }
357 
358       final AbstractJClass thatClassBase = thatClass._extends ();
359       if (thatClassBase != null && this.isAssignableFrom (thatClassBase))
360         return true;
361 
362       final Iterator <AbstractJClass> i = thatClass._implements ();
363       while (i.hasNext ())
364       {
365         final AbstractJClass thatClassInterface = i.next ();
366         if (this.isAssignableFrom (thatClassInterface))
367           return true;
368       }
369       // false so far
370     }
371 
372     return false;
373   }
