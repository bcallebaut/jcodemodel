41  package com.helger.jcodemodel;
42  
43  import java.util.HashSet;
44  
45  import javax.annotation.Nonnull;
46  
47  /**
48   * Utility methods that convert arbitrary strings into Java identifiers.
49   */
50  public final class JJavaName
51  {
52    /** All reserved keywords of Java. */
53    private static final HashSet <String> RESERVED_KEYWORDS = new HashSet <String> ();
54  
55    static
56    {
57      // see
58      // http://java.sun.com/docs/books/tutorial/java/nutsandbolts/_keywords.html
59      final String [] words = new String [] { "abstract",
60                                              "boolean",
61                                              "break",
62                                              "byte",
63                                              "case",
64                                              "catch",
65                                              "char",
66                                              "class",
67                                              "const",
68                                              "continue",
69                                              "default",
70                                              "do",
71                                              "double",
72                                              "else",
73                                              "extends",
74                                              "final",
75                                              "finally",
76                                              "float",
77                                              "for",
78                                              "goto",
79                                              "if",
80                                              "implements",
81                                              "import",
82                                              "instanceof",
83                                              "int",
84                                              "interface",
85                                              "long",
86                                              "native",
87                                              "new",
88                                              "package",
89                                              "private",
90                                              "protected",
91                                              "public",
92                                              "return",
93                                              "short",
94                                              "static",
95                                              "strictfp",
96                                              "super",
97                                              "switch",
98                                              "synchronized",
99                                              "this",
100                                             "throw",
101                                             "throws",
102                                             "transient",
103                                             "try",
104                                             "void",
105                                             "volatile",
106                                             "while",
107 
108                                             // technically these are not
109                                             // reserved
110                                             // words but they cannot be used as
111                                             // identifiers.
112                                             "true",
113                                             "false",
114                                             "null",
115 
116                                             "assert",
117                                             "enum" };
118     for (final String w : words)
119       RESERVED_KEYWORDS.add (w);
120   }
121 
122   private JJavaName ()
123   {}
124 
125   /**
126    * Checks if a given string is usable as a Java identifier.
127    *
128    * @param s
129    *        Source string
130    * @return <code>true</code> if the string is a valid Java identifier
131    */
132   public static boolean isJavaIdentifier (@Nonnull final String s)
133   {
134     if (s.length () == 0)
135       return false;
136     if (RESERVED_KEYWORDS.contains (s))
137       return false;
138 
139     if (!Character.isJavaIdentifierStart (s.charAt (0)))
140       return false;
141 
142     for (int i = 1; i < s.length (); i++)
143       if (!Character.isJavaIdentifierPart (s.charAt (i)))
144         return false;
145 
146     return true;
147   }
148 
149   /**
150    * Checks if the given string is a valid fully qualified name.
151    *
152    * @param sName
153    *        Source string to check
154    * @return <code>true</code> if it is a valid fully qualified class name
155    */
156   public static boolean isFullyQualifiedClassName (final String sName)
157   {
158     return isJavaPackageName (sName);
159   }
160 
161   /**
162    * Checks if the given string is a valid Java package name.
163    *
164    * @param sName
165    *        Source string to check
166    * @return <code>true</code> if it is a valid Java package name
167    */
168   public static boolean isJavaPackageName (final String sName)
169   {
170     String s = sName;
171     while (s.length () != 0)
172     {
173       int idx = s.indexOf ('.');
174       if (idx == -1)
175         idx = s.length ();
176       if (!isJavaIdentifier (s.substring (0, idx)))
177         return false;
178 
179       s = s.substring (idx);
180       if (s.length () != 0)
181       {
182         // remove '.'
183         s = s.substring (1);
184       }
185     }
186     return true;
187   }
188 }
